CREATE TABLE LECTURE(
LECTURE_NO int,
PRIMARY KEY (LECTURE_NO),
START_TIME VARCHAR(10) NOT NULL,
END_TIME VARCHAR(10) );

CREATE TABLE COURSE(
Course_ID INT,
PRIMARY KEY(Course_ID),
Course_Name VARCHAR(15) NOT NULL,
Credit_Hours INT NOT NULL);

CREATE TABLE LOGIN(
Username VARCHAR(8),
PRIMARY KEY(USERNAME),
Password VARCHAR(10) NOT NULL);

CREATE TABLE Student(

RollNo VARCHAR(8),
PRIMARY KEY(RollNo)
Batch INT,
Section VARCHAR(1),

);

CREATE TABLE Student_Course(

RollNo VARCHAR(8),
FOREIGN KEY(RollNo) REFERENCES Student,

CourseID1 INT,
FOREIGN KEY(CourseID1) REFERENCES COURSE,

CourseID2 INT,
FOREIGN KEY(CourseID2) REFERENCES COURSE,

CourseID3 INT,
FOREIGN KEY(CourseID3) REFERENCES COURSE,

CourseID4 INT,
FOREIGN KEY(CourseID4) REFERENCES COURSE,

);
